01#define SLP_EN  (1 << 13)
02void shutdown() // by Napalm and Falkman
03{
04    unsigned int i, j, len, count, found, *ptr, *rsdp = 0, *rsdt = 0, *facp = 0, *dsdt = 0;
05    unsigned char *data, slp_typ[2];
06     
07    // find acpi RSDP table pointer
08    for(ptr = (unsigned int *)0x000E0000; ptr < (unsigned int *)0x000FFFFF; ptr++){
09        if(*ptr == ' DSR' && *(ptr + 1) == ' RTP'){ // "RSD PTR "
10            if(crc8((unsigned char *)ptr, 20)) continue;
11            rsdp = ptr;
12            break;
13        }
14    }
15    if(!rsdp) goto haltonly;
16     
17    // find RSDT table pointer
18    ptr = (unsigned int *)*(ptr + 4);
19    if(crc8((unsigned char *)ptr, *(ptr + 1))){
20        printf("Error: RSDT checksum mismatch.\n");
21        goto haltonly;
22    }
23    rsdt  = ptr;
24    count = (*(ptr + 1) - 36) / 4;
25     
26    // find FACP table pointer
27    ptr += 9; // skip RSDT entries
28    for(i = 0; i < count; i++){
29        for(j = 0; j < 24; j++){
30            if(*(unsigned int *)*ptr == 'PCAF'){ // "FACP"
31                facp = (unsigned int *)*ptr;
32                i = count;
33                break;
34            }
35        }
36    }
37    if(!facp){
38        printf("Error: Could not find FACP table.\n");
39        goto haltonly;
40    }
41    if(crc8((unsigned char *)facp, *(facp + 1))){
42        printf("Error: FACP checksum mismatch.\n");
43        goto haltonly;
44    }
45     
46    // find DSDT table pointer
47    ptr = (unsigned int *)*(facp+10); // DSDT address
48    if(*ptr != 'TDSD'){ // "DSDT"
49        printf("Error: Could not find DSDT table.\n");
50        goto haltonly;
51    }
52    if(crc8((unsigned char *)ptr, *(ptr + 1))){
53        printf("Error: DSDT checksum mistmatch.\n");
54        goto haltonly;
55    }
56    dsdt = ptr;
57     
58    // Search DSDT byte-code for ACPI _S5 signature
59    found = 0;
60    len   = *(dsdt + 1) - 36;
61    data  = (unsigned char *)(dsdt + 36);
62    while(len--){
63        if((*(unsigned int *)data & 0x00FFFFFF) == 0x0035535F){ // "_S5"
64            data += 4;
65            if(*data == 0x12){ // 0x012 = package opcode
66                data += 3; // 0x0A = 8bit integer opcode
67                slp_typ[0] = (*data == 0x0A) ? *++data : *data;
68                data++;
69                slp_typ[1] = (*data == 0x0A) ? *++data : *data;
70                found = 1;
71            }
72            break;
73        }
74        data++;
75    }
76    if(!found) goto haltonly;
77     
78    // execute the actual shutdown and power-off
79    outw(*(facp + 16), slp_typ[0] | SLP_EN);     // FACP[64] = PM1a_CNT_BLK
80    if(*(facp + 17))
81        outw(*(facp + 17), slp_typ[1] | SLP_EN); // FACP[68] = PM1b_CNT_BLK
82    printf("Shutting down...\n");
83    halt();
84     
85haltonly:
86    printf("It is now safe to turn-off your computer.\n");
87    halt();
88     
89}

The function Napalm posted would obviously kick you off for starters.

If you're serious though, I would think of Napalm and falkman's function as a good example starting point for an ACPI driver. The function posted makes good use of one of the facilities made avaliable by the ACPI - there are many others: hibernation, sleep, wake, battery power level detection etc...

An ACPI driver is definitely not one of immediates on your operating system todo list though (the above function will do!). This post is mainly because I did find it useful (when I was playing with OSDev) knowing how everything links together - and you can apply that knowledge to your understanding of how the much bigger operating systems that we use work. Many of the guys on IRC will guide you through the early stages of OSDev and let you know what stages are next, which is incredibly useful.
